<script>
    // Örnek veri seti
    const boxes = [
        { id: "RFID-TX-101", name: "Kutu #1", hasWeapon: false, type: "Tabanca" },
        { id: "RFID-TX-102", name: "Kutu #2", hasWeapon: false, type: "Tüfek" },
        { id: "RFID-TX-103", name: "Kutu #3", hasWeapon: false, type: "Mühimmat" },
        { id: "RFID-TX-104", name: "Kutu #4", hasWeapon: false, type: "Tabanca" },
        { id: "RFID-TX-105", name: "Kutu #5", hasWeapon: false, type: "Tüfek" },
        { id: "RFID-TX-106", name: "Kutu #6", hasWeapon: false, type: "Ekipman" }
    ];

    function renderInventory() {
        const grid = document.getElementById('inventory');
        grid.innerHTML = '';
        
        let filledCount = 0;

        boxes.forEach(box => {
            if(box.hasWeapon) filledCount++;
            
            const card = document.createElement('div');
            card.className = `box-card ${box.hasWeapon ? 'active' : ''}`;
            card.innerHTML = `
                <h4>${box.name}</h4>
                <p><small>ID: ${box.id}</small></p>
                <p>Kategori: ${box.type}</p>
                <span class="status-badge ${box.hasWeapon ? 'filled' : 'empty'}">
                    ${box.hasWeapon ? 'Silah Mevcut' : 'Boş'}
                </span>
            `;
            grid.appendChild(card);
        });

        document.getElementById('filled-boxes').innerText = filledCount;
        document.getElementById('empty-boxes').innerText = boxes.length - filledCount;
    }

    function simulateScan() {
        // Rastgele kutulara silah yerleştirme simülasyonu
        boxes.forEach(box => {
            box.hasWeapon = Math.random() > 0.5; // %50 ihtimalle dolu/boş
        });
        
        const btn = document.querySelector('.scan-btn');
        btn.innerText = "Taranıyor...";
        btn.disabled = true;

        setTimeout(() => {
            renderInventory();
            btn.innerText = "RFID Taraması Başlat";
            btn.disabled = false;
        }, 800);
    }

    // İlk açılışta render et
    renderInventory();
</script>
